add_namespace = superevents

country_event = {
	id = superevents.504
	title = superevents.504.t
	desc = superevents.504.d
	
	immediate = {
		set_global_flag = SE_superevents_activate
		set_global_flag = SE_1st_patsaniwar
		
		#cue funni music
		sound_effect = superevents_1st_patsaniwar
	}

	hidden = yes
	
	fire_only_once = yes

	trigger = {
		NOT = {
			has_global_flag = SE_superevents_activate
		}
		VIK = {
			has_war = yes
		}
		any_country = {
			has_war_with = VIK
			NOT = {
				original_tag = VIK
			}
		}
		NOT = {
			has_global_flag = SE_superevents_activate
		}
		
	}
	option = {
		name = superevents.0.a
	}
}
